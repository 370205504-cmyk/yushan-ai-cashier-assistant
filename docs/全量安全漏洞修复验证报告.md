  try {
    await connection.beginTransaction();
    const result = await callback(connection);
    await connection.commit();
    return result;

| **验证方式** | CSRF Token验证 |

**代码位置**: [middleware/security.js#L145-L159](file:///workspace/lambda/middleware/security.js#L145-L159)

```javascript
const csrfProtection = (req, res, next) => {
  if (['GET', 'HEAD', 'OPTIONS'].includes(req.method)) {
    return next();
  }
  const token = req.headers['x-csrf-token'] || req.body._csrf;
  const sessionToken = req.session?.csrfToken;
  if (process.env.NODE_ENV === 'production' && sessionToken && token !== sessionToken) {
    return res.status(403).json({ success: false, message: '无效的请求令牌' });
  }
  next();
};
```
